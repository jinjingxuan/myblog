# commanderе’ҢInquirer

жң¬ж–Үдё»иҰҒд»Ӣз»Қ node е‘Ҫд»ӨиЎҢзҡ„дәӨдә’е·Ҙе…· commanderпјҲеӨ„зҗҶе‘Ҫд»ӨиЎҢеҸӮж•°пјү е’Ң inquirerпјҲеӨ„зҗҶй—®йўҳдәӨдә’пјү

# commander

е®Ңж•ҙзҡ„ node.js е‘Ҫд»ӨиЎҢи§ЈеҶіж–№жЎҲпјҡе®ҳж–№ж–ҮжЎЈ (opens new window)

# program еҸҳйҮҸ

дёәз®ҖеҢ–дҪҝз”ЁпјҢCommander жҸҗдҫӣдәҶдёҖдёӘе…ЁеұҖеҜ№иұЎ program

const { program } = require('commander');

еҰӮжһңзЁӢеәҸиҫғдёәеӨҚжқӮпјҢз”ЁжҲ·йңҖиҰҒд»ҘеӨҡз§Қж–№ејҸжқҘдҪҝз”Ё CommanderпјҢеҰӮеҚ•е…ғжөӢиҜ•зӯүгҖӮеҲӣе»әжң¬ең°CommandеҜ№иұЎжҳҜдёҖз§ҚжӣҙеҘҪзҡ„ж–№ејҸпјҡ

const { Command } = require('commander');
const program = new Command();

# version

дҪңз”Ёпјҡе®ҡд№үе‘Ҫд»ӨзЁӢеәҸзҡ„зүҲжң¬еҸ· з”Ёжі•зӨәдҫӢпјҡ.version('0.0.1', '-v, --version') еҸӮж•°и§Јжһҗпјҡ

  1. з¬¬дёҖдёӘеҸӮж•° зүҲжң¬еҸ·<еҝ…йЎ»>
  2. з¬¬дәҢдёӘеҸӮж•° иҮӘе®ҡд№үж Үеҝ—<еҸҜзңҒз•Ҙ>пјҡй»ҳи®Өдёә -V е’Ң --version

# option

Commander дҪҝз”Ё.option()ж–№жі•жқҘе®ҡд№үйҖүйЎ№пјҢеҗҢж—¶еҸҜд»Ҙйҷ„еҠ йҖүйЎ№зҡ„з®Җд»ӢгҖӮжҜҸдёӘйҖүйЎ№еҸҜд»Ҙе®ҡд№үдёҖдёӘзҹӯйҖүйЎ№еҗҚз§°пјҲ-еҗҺйқўжҺҘеҚ•дёӘеӯ—з¬Ұпјүе’ҢдёҖдёӘй•ҝйҖүйЎ№еҗҚз§°пјҲ--еҗҺйқўжҺҘдёҖдёӘжҲ–еӨҡдёӘеҚ•иҜҚпјүпјҢдҪҝз”ЁйҖ—еҸ·гҖҒз©әж јжҲ–|еҲҶйҡ”гҖӮ

boolean еһӢйҖүйЎ№ гҖҒеёҰеҸӮж•°йҖүйЎ№

жңүдёӨз§ҚжңҖеёёз”Ёзҡ„йҖүйЎ№пјҢдёҖзұ»жҳҜ boolean еһӢйҖүйЎ№пјҢйҖүйЎ№ж— йңҖй…ҚзҪ®еҸӮж•°пјҢеҸҰдёҖзұ»йҖүйЎ№еҲҷеҸҜд»Ҙи®ҫзҪ®еҸӮж•°пјҲдҪҝз”Ёе°–жӢ¬еҸ·еЈ°жҳҺеңЁиҜҘйҖүйЎ№еҗҺпјҢеҰӮ--expect <value>пјүгҖӮеҰӮжһңеңЁе‘Ҫд»ӨиЎҢдёӯдёҚжҢҮе®ҡе…·дҪ“зҡ„йҖүйЎ№еҸҠеҸӮж•°пјҢеҲҷдјҡиў«е®ҡд№үдёәundefinedгҖӮз¬¬дёүдёӘеҸӮж•°дёәй»ҳи®ӨеҖјгҖӮ

// test.js
program
  .version('0.0.1', '-v, --version')
  .option('-d, --debug', 'output extra debugging')
  .option('-s, --small', 'small pizza size')
  .option('-p, --pizza-type <type>', 'flavour of pizza');

// и§Јжһҗ process.argv
program.parse(process.argv);

if (program.debug) console.log(program.opts());
if (program.small) console.log('- small pizza size');
if (program.pizzaType) console.log(`- ${program.pizzaType}`);
