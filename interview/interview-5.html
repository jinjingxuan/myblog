# д»Җд№ҲжҳҜдәӢд»¶еҶ’жіЎпјҢдәӢд»¶жҚ•иҺ·пјҢдәӢд»¶е§”жүҳ

  • дәӢд»¶еҶ’жіЎ

еҪ“з»ҷзҲ¶еӯҗе…ғзҙ зҡ„еҗҢдёҖдәӢд»¶з»‘е®ҡж–№жі•зҡ„ж—¶еҖҷпјҢи§ҰеҸ‘еӯҗе…ғзҙ иә«дёҠзҡ„дәӢд»¶пјҢжү§иЎҢе®ҢжҜ•д№ӢеҗҺпјҢд№ҹдјҡи§ҰеҸ‘зҲ¶зә§е…ғзҙ зӣёеҗҢзҡ„дәӢд»¶пјҢиҝҷз§ҚжңәеҲ¶еҸ«дәӢд»¶еҶ’жіЎ

  • дәӢд»¶жҚ•иҺ·

з»ҷзҲ¶еӯҗе…ғзҙ з”Ёз»‘е®ҡеҗҢдёҖдәӢд»¶ж—¶пјҢеҪ“и§ҰеҸ‘еӯҗе…ғзҙ иә«дёҠзҡ„дәӢд»¶ж—¶пјҢе…Ҳи§ҰеҸ‘зҲ¶е…ғзҙ пјҢз„¶еҗҺеңЁдј йҖ’з»ҷеӯҗе…ғзҙ пјҢиҝҷз§Қдј ж’ӯжңәеҲ¶еҸ«дәӢд»¶жҚ•иҺ·пјӣ

е®һйҷ…ж“ҚдҪңдёӯпјҢжҲ‘д»¬еҸҜд»ҘйҖҡиҝҮ element.addEventListener() и®ҫзҪ®дёҖдёӘе…ғзҙ зҡ„дәӢд»¶жЁЎеһӢдёәеҶ’жіЎдәӢд»¶жҲ–иҖ…жҚ•иҺ·дәӢд»¶гҖӮ е…ҲжқҘзңӢдёҖдёӢ addEventListener еҮҪж•°зҡ„иҜӯжі•пјҡ

element.addEventListener(type, listener, useCapture)
  • type зӣ‘еҗ¬дәӢд»¶зұ»еһӢзҡ„еӯ—з¬ҰдёІ
  • listener дәӢд»¶зӣ‘еҗ¬еӣһи°ғеҮҪж•°пјҢеҚідәӢд»¶и§ҰеҸ‘еҗҺиҰҒеӨ„зҗҶзҡ„еҮҪж•°
  • useCapture й»ҳи®ӨеҖјfalseпјҢиЎЁзӨәдәӢд»¶еҶ’жіЎпјӣи®ҫдёәtrueж—¶пјҢиЎЁзӨәдәӢд»¶жҚ•иҺ·

  • дәӢд»¶е§”жүҳ

дәӢд»¶е§”жүҳе°ұжҳҜеҸӘжҢҮе®ҡдёҖдёӘдәӢд»¶еӨ„зҗҶзЁӢеәҸпјҢе°ұеҸҜд»Ҙз®ЎзҗҶжҹҗдёҖзұ»еһӢзҡ„жүҖжңүдәӢд»¶гҖӮ

дёәд»Җд№ҲиҰҒз”ЁдәӢд»¶е§”жүҳ

# д»Җд№ҲжҳҜеӣһжөҒ

еҪ“render treeдёӯзҡ„дёҖйғЁеҲҶ(жҲ–е…ЁйғЁ)еӣ дёәе…ғзҙ зҡ„и§„жЁЎе°әеҜёпјҢеёғеұҖпјҢйҡҗи—Ҹзӯүж”№еҸҳиҖҢйңҖиҰҒйҮҚж–°жһ„е»әгҖӮиҝҷе°ұз§°дёәеӣһжөҒ(reflow)гҖӮжҜҸдёӘйЎөйқўиҮіе°‘йңҖиҰҒдёҖж¬ЎеӣһжөҒпјҢе°ұжҳҜеңЁйЎөйқўз¬¬дёҖж¬ЎеҠ иҪҪзҡ„ж—¶еҖҷпјҢиҝҷж—¶еҖҷжҳҜдёҖе®ҡдјҡеҸ‘з”ҹеӣһжөҒзҡ„пјҢеӣ дёәиҰҒжһ„е»әrender treeгҖӮеңЁеӣһжөҒзҡ„ж—¶еҖҷпјҢжөҸи§ҲеҷЁдјҡдҪҝжёІжҹ“ж ‘дёӯеҸ—еҲ°еҪұе“Қзҡ„йғЁеҲҶеӨұж•ҲпјҢе№¶йҮҚж–°жһ„йҖ иҝҷйғЁеҲҶжёІжҹ“ж ‘пјҢе®ҢжҲҗеӣһжөҒеҗҺпјҢжөҸи§ҲеҷЁдјҡйҮҚж–°з»ҳеҲ¶еҸ—еҪұе“Қзҡ„йғЁеҲҶеҲ°еұҸе№•дёӯпјҢиҜҘиҝҮзЁӢжҲҗдёәйҮҚз»ҳгҖӮ

# д»Җд№ҲжҳҜйҮҚз»ҳ

еҪ“render treeдёӯзҡ„дёҖдәӣе…ғзҙ йңҖиҰҒжӣҙж–°еұһжҖ§пјҢиҖҢиҝҷдәӣеұһжҖ§еҸӘжҳҜеҪұе“Қе…ғзҙ зҡ„еӨ–и§ӮпјҢйЈҺж јпјҢиҖҢдёҚдјҡеҪұе“ҚеёғеұҖзҡ„пјҢжҜ”еҰӮbackground-colorгҖӮеҲҷе°ұеҸ«з§°дёәйҮҚз»ҳгҖӮ

  • еҪ“йЎөйқўеёғеұҖе’ҢеҮ дҪ•еұһжҖ§ж”№еҸҳж—¶е°ұйңҖиҰҒеӣһжөҒгҖӮжҜ”еҰӮпјҡж·»еҠ жҲ–иҖ…еҲ йҷӨеҸҜи§Ғзҡ„DOMе…ғзҙ пјҢе…ғзҙ дҪҚзҪ®ж”№еҸҳпјҢе…ғзҙ е°әеҜёж”№еҸҳвҖ”вҖ”иҫ№и·қгҖҒеЎ«е……гҖҒиҫ№жЎҶгҖҒе®ҪеәҰе’Ңй«ҳеәҰпјҢеҶ…е®№ж”№еҸҳ

ж•°жҚ®еұһжҖ§пјҲpropertyпјүз”ЁдәҺе®һзҺ°JavaScriptеј•ж“ҺпјҢжҳҜеұһжҖ§(property)зҡ„еҶ…йғЁеҖјпјҢе®ғеҢ…еҗ«дёҖдёӘж•°жҚ®еҖјзҡ„дҪҚзҪ®гҖӮжңүеҰӮдёӢ4з§ҚиЎҢдёәзү№жҖ§гҖӮ

  • [[Configurable]]пјҡиғҪеҗҰиў«deleteеҲ йҷӨеұһжҖ§йҮҚж–°е®ҡд№ү
  • [[Enumerable]]пјҡиғҪеҗҰиў«for-inжһҡдёҫ
  • [[Writable]]пјҡиғҪеҗҰдҝ®ж”№еұһжҖ§еҖј
  • [[Value]]пјҡж•°жҚ®зҡ„ж•°жҚ®еҖј
let person = {}
Object.defineProperty(person,"name",{
    writable:false   //дёҚеҸҜдҝ®ж”№еұһжҖ§
    value:"abc"
})

# и®ҝй—®еҷЁеұһжҖ§

и®ҝй—®еҷЁеұһжҖ§дё»иҰҒз”ұsetterе’ҢgetterеҮҪж•°з»„жҲҗпјҢеҢ…еҗ«еҰӮдёӢ4дёӘзү№жҖ§пјҡ

  • [[Configurable]]пјҡиғҪеҗҰиў«deleteеҲ йҷӨеұһжҖ§йҮҚж–°е®ҡд№үгҖӮй»ҳи®ӨеҖјпјҡtrue
  • [[Enumerable]]пјҡиғҪеҗҰиў«for-inжһҡдёҫгҖӮй»ҳи®ӨеҖјпјҡtrue
  • [[Get]]пјҡиҜ»еҸ–еұһжҖ§еҖјгҖӮй»ҳи®ӨеҖјпјҡundefined
  • [[Set]]пјҡеҶҷе…ҘеұһжҖ§еҖјгҖӮй»ҳи®ӨеҖјпјҡundefined
let book = {
    _year:2004, //_yearеүҚйқўзҡ„дёӢеҲ’зәҝжҳҜдёҖз§Қеёёз”Ёзҡ„и®°еҸ·пјҢз”ЁдәҺиЎЁзӨәеҸӘиғҪйҖҡиҝҮеҜ№иұЎж–№жі•и®ҝй—®зҡ„еұһжҖ§гҖӮ
    edition:1
}
Object.defineProperty(book,"year",{
    get:function(){
        return this._year
    },
    set:function(newValue){
        if(newValue>2004){
            this._year = newValue
            this.edition += newValue-2004
        }
    }
})

book.year = 2005
console.log(book.edition)
